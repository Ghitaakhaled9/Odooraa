function updateSubtotal() { // Get the quantity input value var quantity = parseInt(document.getElementById('quantityInput').value); // Get the price per unit var price = parseFloat(document.getElementById('price').innerText); // Calculate the subtotal var subtotal = quantity * price; // Update the subtotal cell with the calculated value document.getElementById('subtotal').innerText = subtotal.toFixed(2); // Update the total amount var totalAmount = document.getElementById('totalAmount'); var currentTotal = parseFloat(totalAmount.innerText.replace('$', '')); var newTotal = currentTotal - parseFloat(document.getElementById('subtotal').innerText) + subtotal; totalAmount.innerText = '$' + newTotal.toFixed(2); }
